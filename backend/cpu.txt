/* ── Theme picker popover ────────────────────────────────────────── */
function ThemePicker({ pickerState, panelThemeIds, onSelect, onClose }) {
  if (!pickerState) return null;
  const { panelId, x, y } = pickerState;

  return (
    <>
      {/* Backdrop */}
      <div onClick={onClose} style={{ position: 'fixed', inset: 0, zIndex: 998 }} />

      {/* Picker panel */}
      <div
        onClick={e => e.stopPropagation()}
        style={{
          position: 'fixed',
          top: y,
          left: Math.min(x, window.innerWidth - 230),
          zIndex: 999,
          backgroundColor: '#ffffff',
          border: '1px solid #fecaca',
          borderRadius: 16,
          padding: '14px',
          boxShadow: '0 16px 40px rgba(185,28,28,0.14)',
          width: 220,
        }}
      >
        <div style={{
          fontSize: 9.5, fontWeight: 700, color: '#b91c1c',
          textTransform: 'uppercase', letterSpacing: '0.1em', marginBottom: 10,
        }}>
          Choose Theme
        </div>

        <div style={{ display: 'grid', gridTemplateColumns: 'repeat(4, 1fr)', gap: 6 }}>
          {THEME_PRESETS.map(preset => (
            <div
              key={preset.id}
              onClick={() => { onSelect(panelId, preset.id); onClose(); }}
              title={preset.name}
              style={{
                height: 34,
                borderRadius: 8,
                background: preset.gradient,
                cursor: 'pointer',
                border: panelThemeIds[panelId] === preset.id
                  ? '2.5px solid #ffffff'
                  : '2.5px solid transparent',
                boxShadow: panelThemeIds[panelId] === preset.id
                  ? '0 0 0 1px rgba(255,255,255,0.3)'
                  : 'none',
                transition: 'transform 0.15s ease, border-color 0.15s ease',
                position: 'relative',
              }}
              onMouseEnter={e => e.currentTarget.style.transform = 'scale(1.08)'}
              onMouseLeave={e => e.currentTarget.style.transform = 'scale(1)'}
            />
          ))}
        </div>

        <div style={{ fontSize: 9.5, color: '#b91c1c', marginTop: 10, textAlign: 'center' }}>
          {THEME_PRESETS.find(p => p.id === panelThemeIds[panelId])?.name}
        </div>
      </div>
    </>
  );
}
